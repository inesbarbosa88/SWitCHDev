CREATE OR REPLACE TRIGGER trg_validate_email
BEFORE INSERT OR UPDATE ON student
FOR EACH ROW
BEGIN
IF :NEW.email IS NOT NULL AND
NOT REGEXP_LIKE(:NEW.email, '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$') THEN
RAISE_APPLICATION_ERROR(-20001, 'Invalid email format.');
END IF;
END;

-- ## Table Student_Course ##
CREATE TABLE student_course ( 
id_course INTEGER NOT NULL, 
id_student INTEGER NOT NULL, 
final_grade INTEGER, 
attempted_improvement CHAR(1) NOT NULL CHECK (UPPER (attempted_improvement) IN ('Y', 'N')), 
enrollment_datetime TIMESTAMP NOT NULL,
PRIMARY KEY (id_course, id_student),

CONSTRAINT fk_course FOREIGN KEY (id_course) REFERENCES course(id_course),
CONSTRAINT fk_student FOREIGN KEY (id_student) REFERENCES student(id_student),

CONSTRAINT chk_final_grade CHECK (final_grade BETWEEN 0.00 AND 20.00)
);

CREATE OR REPLACE TRIGGER check_attempted_improvement BEFORE INSERT OR UPDATE ON student_course FOR EACH ROW BEGIN

IF :NEW.final_grade IS NULL AND UPPER(:NEW.attempted_improvement) != 'N' THEN
RAISE_APPLICATION_ERROR(-20001, 'If final grade is NULL, attempted improvement must be ''N''.');

ELSIF :NEW.final_grade < 10 AND UPPER(:NEW.attempted_improvement) != 'N' THEN
RAISE_APPLICATION_ERROR(-20002, 'If final grade is less than 10, attempted improvement must be ''N''.');

ELSIF :NEW.final_grade >= 10 AND UPPER (:NEW.attempted_improvement) NOT IN ('Y', 'N') THEN
RAISE_APPLICATION_ERROR(-20003, 'If final grade is 10 or greater, attempted improvement can be ''Y'' or ''N''.');
END IF;
END;
/

CREATE OR REPLACE TRIGGER check_grade_improvement
BEFORE INSERT OR UPDATE ON student_course
FOR EACH ROW
DECLARE
    original_grade INTEGER;
BEGIN
 
    IF :NEW.id_student IS NOT NULL AND :NEW.id_course IS NOT NULL THEN
        BEGIN
           
            SELECT final_grade
            INTO original_grade
            FROM student_course
            WHERE id_student = :NEW.id_student
              AND id_course = :NEW.id_course
              AND final_grade IS NOT NULL
              AND :NEW.enrollment_datetime > enrollment_datetime
            ORDER BY enrollment_datetime DESC
            FETCH FIRST 1 ROWS ONLY; 
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                original_grade := NULL;
        END;

        IF :NEW.attempted_improvement = 'Y' AND :NEW.final_grade < original_grade THEN
            RAISE_APPLICATION_ERROR(-20004, 'The new grade cannot be lower than the original grade when attempting an improvement.');
        END IF;
    END IF;
END;
/

CREATE TABLE classroom (
id_classroom CHAR(4) PRIMARY KEY,
building CHAR(1) NOT NULL,
id_classroom_type INTEGER NOT NULL,
FOREIGN KEY (id_classroom_type) REFERENCES classroom_type(id_classroom_type),
-- Restringe building a A-H
CONSTRAINT chk_building CHECK (building IN ('A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'))
);
CREATE OR REPLACE TRIGGER trg_validate_classroom BEFORE INSERT OR UPDATE ON classroom FOR EACH ROW BEGIN

IF SUBSTR(:NEW.id_classroom, 1, 1) != :NEW.building THEN
RAISE_APPLICATION_ERROR(-20001, 'The first character of id_classroom must match the building.');
END IF;

IF NOT REGEXP_LIKE(SUBSTR(:NEW.id_classroom, 2), '^(10[0-9]|1[1-9][0-9]|[2-4][0-9]{2}|500)$') THEN
RAISE_APPLICATION_ERROR(-20002, 'The numeric part of id_classroom must be between 100 and 500.');
END IF;
END;

--## Table Course_class##
CREATE TABLE course_class ( 
id_course_class INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
date_start_time TIMESTAMP NOT NULL, 
date_end_time TIMESTAMP NOT NULL, 
id_course_edition INTEGER NOT NULL, 
id_classroom CHAR(4) NOT NULL, 
id_teacher INTEGER NOT NULL, 
id_course_class_type INTEGER NOT NULL, 
id_class INTEGER NOT NULL, 
id_weekly_timetable INTEGER NOT NULL,

CONSTRAINT chk_course_class_start_time CHECK (
EXTRACT(MINUTE FROM date_start_time) IN (10, 40)
),

CONSTRAINT chk_course_class_duration CHECK (
EXTRACT(SECOND FROM (date_end_time - date_start_time)) +
EXTRACT(MINUTE FROM (date_end_time - date_start_time)) * 60 +
EXTRACT(HOUR FROM (date_end_time - date_start_time)) * 3600 >= 3599.9
),

CONSTRAINT chk_start_end_time_valid_range CHECK (
EXTRACT(HOUR FROM date_start_time) * 60 + EXTRACT(MINUTE FROM date_start_time) >= 490
AND EXTRACT(HOUR FROM date_start_time) * 60 + EXTRACT(MINUTE FROM date_start_time) <= 1360
AND date_end_time > date_start_time
AND EXTRACT(HOUR FROM date_end_time) * 60 + EXTRACT(MINUTE FROM date_end_time) <= 1420
),

FOREIGN KEY (id_course_edition) REFERENCES course_edition(id_course_edition),
FOREIGN KEY (id_classroom) REFERENCES classroom(id_classroom),
FOREIGN KEY (id_teacher) REFERENCES teacher(id_teacher),
FOREIGN KEY (id_class) REFERENCES class(id_class),
FOREIGN KEY (id_course_class_type) REFERENCES course_class_type(id_course_class_type),
FOREIGN KEY (id_weekly_timetable) REFERENCES weekly_timetable(id_weekly_timetable)
);
